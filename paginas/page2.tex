% -- Estilo visual limpio para la tabla --
\newcommand{\StaticField}[1]{%
  \tcbox[on line,enhanced,
    colback=warningyellow,colframe=warningyellow,
    boxrule=0pt,arc=1mm,
    width=\linewidth,
    left=0mm,right=0mm,top=1.1mm,bottom=1.1mm,
    valign=center,halign=center
  ]{{\fontsize{8.4}{9.8}\selectfont\bfseries #1}}%
}

\newcommand{\ValueField}[1]{%
  \tcbox[on line,enhanced,
    colback=gray!15,colframe=gray!15,
    boxrule=0pt,arc=1mm,
    width=\linewidth,
    left=0mm,right=0mm,top=1.1mm,bottom=1.1mm,
    valign=center,halign=left
  ]{{\fontsize{8.4}{9.8}\selectfont\strut #1}}%
}

% -- Bloque simple editable por posición (sin contenedor) --
% #1 Posición | #2 Logo | #3 Nombre llanta | #4 Medida | #5 PSI | #6 Profundidad | #7 Desgaste discos | #8 Desgaste balatas
\newcommand{\SimpleWheelBlock}[8]{%
  {\centering\bfseries\kodemono #1\par}
  \vspace{0.8mm}

  % Fila superior: logo izquierda + nombre derecha (envuelve texto largo)
  \noindent
  \begin{minipage}[c]{0.22\linewidth}
    \centering
    \includegraphics[width=0.85\linewidth,height=8mm,keepaspectratio]{#2}
  \end{minipage}%
  \hfill
  \begin{minipage}[c]{0.75\linewidth}
    {\raggedright\fontsize{8.2}{9.4}\selectfont\bfseries #3\par}
  \end{minipage}

  \vspace{0.8mm}

  % Tabla limpia: etiquetas estáticas en amarillo, valores variables en gris
  \setlength{\tabcolsep}{0pt}
  \renewcommand{\arraystretch}{1.0}
  \begin{tabular}{p{0.42\linewidth}p{0.58\linewidth}}
    \StaticField{MEDIDA} & \ValueField{#4} \\
    \StaticField{PSI} & \ValueField{#5} \\
    \StaticField{PROFUNDIDAD} & \ValueField{#6} \\
    \StaticField{DESGASTE DISCOS} & \ValueField{#7} \\
    \StaticField{DESGASTE BALATAS} & \ValueField{#8} \\
  \end{tabular}
}
