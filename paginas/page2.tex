\newcommand{\StaticFieldColor}[2]{%
  \tcbox[on line,enhanced,
    colback=#2,colframe=black!25,
    boxrule=0.45pt,arc=1.0mm,
    width=\linewidth,
    left=0mm,right=0mm,top=1.0mm,bottom=1.0mm,
    valign=center,halign=center
  ]{{\fontsize{8.4}{9.8}\selectfont\bfseries #1}}%
}

\newcommand{\ValueFieldColor}[2]{%
  \tcbox[on line,enhanced,
    colback=#2,colframe=black!25,
    boxrule=0.45pt,arc=1.0mm,
    width=\linewidth,
    left=0mm,right=0mm,top=1.0mm,bottom=1.0mm,
    valign=center,halign=center
  ]{{\fontsize{8.4}{9.8}\selectfont #1}}%
}

\newcommand{\SimpleWheelBlock}[8]{%
  {\centering\bfseries\kodemono #1\par}
  \vspace{0.8mm}

  % Fila superior: logo izquierda + nombre derecha (envuelve texto largo)

    \includegraphics[width=0.85\linewidth,height=8mm,keepaspectratio]{#2}
  \end{minipage}%
  \hfill
  \begin{minipage}[c]{0.75\linewidth}
    {\raggedright\fontsize{8.2}{9.4}\selectfont\bfseries #3\par}
  \end{minipage}

  \vspace{0.8mm}

  % Tabla limpia: cada fila con color propio (sin bordes de color)
  \setlength{\tabcolsep}{0pt}
  \renewcommand{\arraystretch}{1.0}
  \begin{tabular}{p{0.50\linewidth}p{0.50\linewidth}}
    \StaticFieldColor{MEDIDA}{warningyellow} & \ValueFieldColor{#4}{gray!15} \\
    \StaticFieldColor{PSI}{orange!55} & \ValueFieldColor{#5}{gray!15} \\

    \StaticFieldColor{DESGASTE BALATAS}{yellow!85!orange} & \ValueFieldColor{#8}{gray!15} \\
  \end{tabular}
