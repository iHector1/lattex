% ============================================================
% COLORES (AJUSTA AQUÍ)
% ============================================================
% Color del fondo del bloque de la imagen/logo (lado izquierdo)
\providecolor{brandblue}{HTML}{0B4AA2}

% Color del fondo del bloque del título (lado derecho)
\providecolor{headgray}{HTML}{E7E7E7}

% Color del texto del título
\providecolor{headtext}{HTML}{1E1E1E}

% ============================================================
% CONTROLES DE TAMAÑO (AJUSTA AQUÍ)
% ============================================================
% Tamaño GENERAL del header:
% - ANCHO total: depende del ancho del contenedor donde llames \TPCardHeader.
% - ALTO total: lo controla \TPHeaderHeight.
% Proporciones horizontales del header (deben sumar 1.00)
\newcommand{\TPHeaderLeftWidth}{0.30\linewidth}   % ancho bloque imagen
\newcommand{\TPHeaderRightWidth}{0.70\linewidth}  % ancho bloque título

% Altura total del header (RESPONSIVA: depende de \textwidth)
\newcommand{\TPHeaderHeight}{0.048\textwidth}

% Tamaño de la imagen/logo (RESPONSIVO)
\newcommand{\TPHeaderImageHeight}{0.022\textwidth}

% Tipografía del título (texto derecho)
\newcommand{\TPHeaderTitleFontSize}{10.2}
\newcommand{\TPHeaderTitleLineHeight}{11.6}

% Espaciado interno del título
\newcommand{\TPHeaderTitlePadLeft}{2.3mm}
\newcommand{\TPHeaderTitlePadRight}{2mm}

% ============================================================
% HEADER: IMAGEN + TÍTULO AL MISMO NIVEL
% ============================================================
% Uso:
% \TPCardHeader{Goodyear.png}{WRANGLER ALL TERRAIN ADVENTURE W/KEVLAR 110T}
\newcommand{\TPCardHeader}[2]{%
  \noindent
  \begin{tcolorbox}[
    enhanced,
    colback=white, colframe=white,
    boxrule=0pt, arc=0mm,
    left=0mm, right=0mm, top=0mm, bottom=0mm,
    boxsep=0mm
  ]
    % BLOQUE IZQUIERDO (IMAGEN)
    \begin{minipage}[c][\TPHeaderHeight][c]{\TPHeaderLeftWidth}
      \tcbox[
        enhanced,
        colback=brandblue, colframe=brandblue,
        boxrule=0pt, arc=0mm,
        left=0mm, right=0mm, top=0mm, bottom=0mm,
        width=\linewidth,
        height=\TPHeaderHeight,
        valign=center, halign=center
      ]{\includegraphics[height=\TPHeaderImageHeight]{#1}}%
    \end{minipage}%
    % BLOQUE DERECHO (TÍTULO)
    \begin{minipage}[c][\TPHeaderHeight][c]{\TPHeaderRightWidth}
      \tcbox[
        enhanced,
        colback=headgray, colframe=headgray,
        boxrule=0pt, arc=0mm,
        left=\TPHeaderTitlePadLeft, right=\TPHeaderTitlePadRight,
        top=0mm, bottom=0mm,
        width=\linewidth,
        height=\TPHeaderHeight,
        valign=center,
        halign=left
      ]{%
        {\color{headtext}\fontsize{\TPHeaderTitleFontSize}{\TPHeaderTitleLineHeight}\selectfont\bfseries #2}%
      }%
    \end{minipage}%
  \end{tcolorbox}%
}

\endinput
