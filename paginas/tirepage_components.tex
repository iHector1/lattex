\newcommand{\TFHeader}[2]{%
  \noindent
  \begin{minipage}[c][\TFHeaderH][c]{0.34\linewidth}
    \tcbox[
      enhanced,
      colback=TFBlue,colframe=TFBlue,boxrule=0pt,arc=0mm,
      width=\linewidth,height=\TFHeaderH,
      left=0mm,right=0mm,top=0mm,bottom=0mm,
      valign=center,halign=center
    ]{\includegraphics[height=\TFHeaderLogoH,trim=6 3 6 3,clip]{#1}}%
  \end{minipage}%
  \begin{minipage}[c][\TFHeaderH][c]{0.66\linewidth}
    \tcbox[
      enhanced,
      colback=TFHeadGray,colframe=TFHeadGray,boxrule=0pt,arc=0mm,
      width=\linewidth,height=\TFHeaderH,
      left=0.6mm,right=0.6mm,top=0mm,bottom=0mm,
      valign=center,halign=left
    ]{{\fontsize{\TFHeaderFont}{\TFHeaderLead}\selectfont\bfseries\color{TFText} #2}}%
  \end{minipage}%
}
